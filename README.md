# 1° - Caso não tenha o YARN instalado, gentileza instala-lo:

npm install --global yarn

# 2° - Executar o comando no teminal, dentro da pasta da aplicação

yarn compose ou docker-compose up --build -d

# 3° Acessar

http://localhost:8080

# 4° Debug

yarn start => para debug
